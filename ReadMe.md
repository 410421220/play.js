# usage

          var play = require('./lib/play');

          play.sound('./wavs/intro.wav');
          play.sound('./wavs/alarm.wav');
          play.sound('./wavs/crinkle.wav');
          play.sound('./wavs/flush.wav');
          play.sound('./wavs/ding.wav');
