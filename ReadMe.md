# usage

          var play = require('./lib/play');

          play.sound('./wavs/kick.wav');
          play.sound('./wavs/tick.wav');
          play.sound('./wavs/snare.wav');

          // drum party, now go JSONloop => http://github.com/marak/JSONloops
